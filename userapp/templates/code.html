model1_two_zone_fractal.py 
import numpy as np
import matplotlib.pyplot as plt
from scipy.special import gamma
import csv

C0 = 0.1;  vm = 1e-4;  Dm = 1e-5;  n = 20
tau = 1;   h = 0.1;    tmax = 2700; w = 1e-6
tetim = 0.1;  tetm = 0.4;  gam = 0.06
alpha = 0.9;  bet = 2
A = 1 + (w * gamma(2 - alpha) * tau**alpha) / (gam * tetim)

Cm  = np.zeros((tmax + 1, n + 1))
Cim = np.zeros((tmax + 1, n + 1))
Cm2 = np.zeros(tmax + 1);  Cm3 = np.zeros(tmax + 1);  Cm4 = np.zeros(tmax + 1)
Cim2 = np.zeros(tmax + 1); Cim3 = np.zeros(tmax + 1); Cim4 = np.zeros(tmax + 1)

for k in range(tmax + 1):
    Cm[k, 0] = C0

for k in range(tmax):
    for i in range(n + 1):
        s1 = 0
        for l1 in range(k):
            s1 += (Cim[l1+1,i] - Cim[l1,i]) * ((k-l1+1)**(1-alpha) - (k-l1)**(1-alpha))
        Cim[k+1,i] = (Cim[k,i] - s1 + (gamma(2-alpha)*tau**alpha*w)*Cm[k,i]/(gam*tetim)) / A

    for i in range(1, n):
        if bet == 2:
            s01 = Cm[k,i+1] - 2*Cm[k,i] + Cm[k,i-1]
        else:
            s01 = sum(((l+1)**(2-bet) - l**(2-bet)) *
                      (Cm[k,i+1-l] - 2*Cm[k,i-l] + Cm[k,i-1-l]) for l in range(i))
        Cm[k+1,i] = (
            (tau*Dm*s01)/(gamma(3-bet)*h**bet)
            - tau*vm*(Cm[k,i] - Cm[k,i-1])/h
            - gam*tau*tetim*s1/(tetm*gamma(2-alpha)*tau**alpha) * (Cim[k+1,i] - Cim[k,i])
            + Cm[k,i]
        )
    Cm[k+1, n] = Cm[k+1, n-1]

x = [i*h for i in range(n+1)]
y = [k*tau for k in range(tmax+1)]
for k in range(tmax+1):
    Cm2[k] = Cm[k,2];  Cm3[k] = Cm[k,3];  Cm4[k] = Cm[k,4]
    Cim2[k] = Cim[k,2]; Cim3[k] = Cim[k,3]; Cim4[k] = Cim[k,4]

# Профили по x
fig, axes = plt.subplots(1, 2, figsize=(14, 5))
for ax, data, name in [(axes[0], Cm, 'Cm'), (axes[1], Cim, 'Cim')]:
    ax.plot(x, data[tmax,:],       label=f'{name}(t={tmax} s)')
    ax.plot(x, data[2*tmax//3,:],  label=f'{name}(t={2*tmax//3} s)')
    ax.plot(x, data[tmax//3,:],    label=f'{name}(t={tmax//3} s)')
    ax.set_xlabel('X (m)');  ax.set_ylabel('Концентрация')
    ax.legend();  ax.grid(True)
plt.tight_layout();  plt.show()

# Динамика по времени
fig, axes = plt.subplots(1, 2, figsize=(14, 5))
for ax, d2, d3, d4, name in [(axes[0], Cm2, Cm3, Cm4, 'Cm'), (axes[1], Cim2, Cim3, Cim4, 'Cim')]:
    ax.plot(y, d2, label=f'{name}(x=0.1)');  ax.plot(y, d3, label=f'{name}(x=0.3)')
    ax.plot(y, d4, label=f'{name}(x=0.5)');  ax.set_xlabel('t (s)')
    ax.legend();  ax.grid(True)
plt.tight_layout();  plt.show()

# CSV export
for fname, data in [('Cm.csv', Cm[tmax,:]), ('Cim.csv', Cim[tmax,:]),
                    ('C2.csv', Cm2), ('C3.csv', Cm3), ('C4.csv', Cm4)]:
    with open(fname, 'w', newline='') as f:
        csv.writer(f, delimiter=';').writerow(list(data))
model2_adsorption.py 
import numpy as np
import matplotlib.pyplot as plt
from scipy.special import gamma
import csv

C0 = 0.1;  vm = 1e-4;  Dm = 5e-5;  n = 20
tau = 1;   h = 0.1;    tmax = 2700; w = 1e-5
tetim = 0.1;  tetm = 0.4;  gam = 0.6
alpha = 0.9;  bet = 0.9
f = 0.7;  pb = 1000;  kd = 1e-4
mlim = 1e-6;  msim = 1e-8;  mlm = 1e-3;  msm = 1e-4

Rm  = tetm  + f*pb*kd
Rim = tetim + (1-f)*pb*kd
A1  = tetm*mlm  + f*pb*kd*msm
A2  = tetim*mlim + (1-f)*pb*kd*msim
F1  = 1 / (gamma(3-(bet+1)) * h**(bet+1) * Rm)

Cm  = np.zeros((tmax+1, n+1));  Cim = np.zeros((tmax+1, n+1))
Cm2 = np.zeros(tmax+1);         Cm3 = np.zeros(tmax+1);  Cm4 = np.zeros(tmax+1)
Cim2 = np.zeros(tmax+1);        Cim3 = np.zeros(tmax+1); Cim4 = np.zeros(tmax+1)

for k in range(tmax+1):
    Cm[k,0] = C0

for k in range(tmax):
    for i in range(n+1):
        s1 = sum((Cim[l+1,i]-Cim[l,i])*((k-l+1)**(1-alpha)-(k-l)**(1-alpha)) for l in range(k))
        Cim[k+1,i] = Cim[k,i]-s1 + (gamma(2-alpha)*tau**alpha/Rim)*(w*(Cm[k,i]-Cim[k,i])-A2*Cim[k,i])
    for i in range(1, n):
        if (bet+1) == 2:
            s01 = Cm[k,i+1] - 2*Cm[k,i] + Cm[k,i-1]
        else:
            s01 = sum(((l+1)**(2-(bet+1))-l**(2-(bet+1))) *
                      (Cm[k,i+1-l]-2*Cm[k,i-l]+Cm[k,i-1-l]) for l in range(i))
        Cm[k+1,i] = (
            tau*tetm*Dm*F1*s01
            - tau*vm*tetm*(Cm[k,i+1]-Cm[k,i-1])/(Rm*2*h)
            - (tau*w/Rm)*(Cm[k,i]-Cim[k,i])
            - (tau*A1/Rm)*Cm[k,i] + Cm[k,i]
        )
    Cm[k+1,n] = Cm[k+1,n-1]

# Plots and CSV (same pattern as Model 1)
x = [i*h for i in range(n+1)]
y = [k*tau for k in range(tmax+1)]
for k in range(tmax+1):
    Cm2[k]=Cm[k,2]; Cm3[k]=Cm[k,3]; Cm4[k]=Cm[k,4]
    Cim2[k]=Cim[k,2]; Cim3[k]=Cim[k,3]; Cim4[k]=Cim[k,4]

fig, axes = plt.subplots(1,2,figsize=(14,5))
for ax, data, name in [(axes[0],Cm,'Cm'),(axes[1],Cim,'Cim')]:
    ax.plot(x,data[tmax,:],label=f'{name}(t={tmax})'); ax.legend(); ax.grid(True)
plt.tight_layout(); plt.show()
model3_multiterm.py 
import numpy as np
import matplotlib.pyplot as plt
from scipy.special import gamma
import csv

C0 = 0.1;  Dm = 1e-4;  n = 20
tau = 0.5; h = 0.01;   tmax = 900
gam = 2;   alpha = 1;  betta1 = 0.7;  betta2 = 0.5
r1 = 1.5;  r2 = 1.0

T1 = 1 / (gamma(3-gam) * h**gam)
T2 = 1/gamma(2-alpha) * tau**alpha
T3 = r1/gamma(2-betta1) * tau**betta1
T4 = r2/gamma(2-betta2) * tau**betta2

C  = np.zeros((tmax+1, n+1))
C2 = np.zeros(tmax+1);  C3 = np.zeros(tmax+1);  C4 = np.zeros(tmax+1)

for k in range(tmax+1):
    C[k,0] = C0

for k in range(tmax):
    for i in range(1, n):
        s1 = s2 = s3 = 0
        for l1 in range(k):
            diff = C[l1+1,i] - C[l1,i]
            s1 += diff * ((k-l1+1)**(1-alpha)  - (k-l1)**(1-alpha))
            s2 += diff * ((k-l1+1)**(1-betta1) - (k-l1)**(1-betta1))
            s3 += diff * ((k-l1+1)**(1-betta2) - (k-l1)**(1-betta2))
        if gam == 2:
            s01 = C[k,i+1] - 2*C[k,i] + C[k,i-1]
        else:
            s01 = sum((C[k,i-1-l]-2*C[k,i-l]+C[k,i+1-l])*((l+1)**(2-gam)-l**(2-gam))
                      for l in range(i))
        C[k+1,i] = (Dm*T1*s01 - T2*(s1-C[k,i]) - T3*(s2-C[k,i]) - T4*(s3-C[k,i])) / (T2+T3+T4)
    C[k+1,n] = C[k+1,n-1]

x = [i*h for i in range(n+1)]
y = [k*tau for k in range(tmax+1)]
for k in range(tmax+1):
    C2[k]=C[k,2]; C3[k]=C[k,3]; C4[k]=C[k,4]

fig, axes = plt.subplots(1,2,figsize=(14,5))
axes[0].plot(x,C[tmax,:],label=f'C(t={tmax})'); axes[0].legend(); axes[0].grid(True)
axes[1].plot(y,C2,label='x=0.1'); axes[1].plot(y,C3,label='x=0.3')
axes[1].plot(y,C4,label='x=0.5'); axes[1].legend(); axes[1].grid(True)
plt.tight_layout(); plt.show()
